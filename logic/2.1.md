2) Somar dois números

INÍCIO
  ESCREVA("Digite A: ")
  LEIA(a)
  ESCREVA("Digite B: ")
  LEIA(b)
  soma <- a + b
  ESCREVA("Soma = ", soma)
FIM

3) Contar de 1 até 10 usando loop

INÍCIO
  PARA i <- 1 ATÉ 10 FAÇA
    ESCREVA(i)
  FIMPARA
FIM

4) Verificar se o número é par ou ímpar

INÍCIO
  ESCREVA("Digite um inteiro: ")
  LEIA(n)
  SE (n % 2 = 0) ENTÃO
    ESCREVA("PAR")
  SENÃO
    ESCREVA("ÍMPAR")
  FIMSE
FIM

5) Verificar se número é positivo, negativo ou zero

INÍCIO
  ESCREVA("Digite um número: ")
  LEIA(n)
  SE (n > 0) ENTÃO
    ESCREVA("POSITIVO")
  SENÃO SE (n < 0) ENTÃO
    ESCREVA("NEGATIVO")
  SENÃO
    ESCREVA("ZERO")
  FIMSE
FIM

6) Calcular média de 3 notas

INÍCIO
  ESCREVA("Digite as três notas: ")
  LEIA(n1, n2, n3)
  media <- (n1 + n2 + n3) / 3
  ESCREVA("Média = ", media)
FIM

7) Verificar se aluno foi aprovado (média ≥ 5)

INÍCIO
  ESCREVA("Digite as três notas: ")
  LEIA(n1, n2, n3)
  media <- (n1 + n2 + n3) / 3
  SE (media >= 5) ENTÃO
    ESCREVA("APROVADO (Média = ", media, ")")
  SENÃO
    ESCREVA("REPROVADO (Média = ", media, ")")
  FIMSE
FIM

8) Maior entre dois números

INÍCIO
  ESCREVA("Digite A e B: ")
  LEIA(a, b)
  SE (a > b) ENTÃO
    ESCREVA("Maior = ", a)
  SENÃO SE (b > a) ENTÃO
    ESCREVA("Maior = ", b)
  SENÃO
    ESCREVA("São iguais: ", a, " e ", b)
  FIMSE
FIM
